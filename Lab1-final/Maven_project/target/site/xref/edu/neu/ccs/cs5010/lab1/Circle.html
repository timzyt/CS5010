3   /**
4    * This class represents a circle.  It offers all the operations mandated by the Shape interface.
5    */
6   public class Circle extends AbstractShape {
7   
8     private double radius;
9   
10    /**
11     * Construct a circle object using the given center and radius.
12     *
13     * @param xcord x coordinate of the center of this circle
14     * @param ycord y coordinate of the center of this circle
15     * @param radius the radius of this circle
16     */
17    public Circle(double xcord, double ycord, double radius) {
18      super(new Point2D(xcord, ycord));
19      this.radius = radius;
20    }
21  
22    /**
23     * Construct a circle object with the given radius. It is centered at (0,0)
24     *
25     * @param radius the radius of this circle
26     */
27    public Circle(double radius) {
28      super(new Point2D(0, 0));
29      this.radius = radius;
30    }
31  
32    //@Override
33    public double area() {
34      return Math.PI * radius * radius;
35    }
36  
37    @Override
38    public double perimeter() {
39      return 2 * Math.PI * radius;
40    }
41  
42    @Override
43    public Shape resize(double factor) {
44      return new Circle(reference.getX(), reference.getY(), Math.sqrt(factor) * radius);
45    }
46  
47  
48    public String toString() {
49      return String.format("Circle: center (%.3f,%.3f) radius %.3f",
50          this.reference.getX(), this.reference.getY(), this.radius);
51    }
52  }
