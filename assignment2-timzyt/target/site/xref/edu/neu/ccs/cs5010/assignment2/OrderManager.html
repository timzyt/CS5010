41    /**
42     * take a SKU# as an argument and add that item to the current order if that item is in stock. It
43     * should give useful feedback by standard output indicating whether the item was added to the
44     * order (a SKU# that is not in stock should not be added, and the user should be informed of the
45     * issue).
46     *
47     * @param newSku the new sku
48     */
49    public void addToOrder(String newSku) {
50      if (newSku == null) {
51        throw new NullValueException("Provided SKU# is null.");
52      }
53      if (newSku.length() == 0) {
54        throw new InvalidConstructorArgumentsException("Provided SKU# is invalid.");
55      }
56  
57      if (theStockService.getStock().containsKey(newSku)) {
58        if (this.order.containsKey(newSku) || this.order.get(newSku) != null) {
59          Integer newCount = Integer.valueOf(this.order.get(newSku).intValue() + 1);
60          this.order.put(newSku, newCount);
61        } else {
62          this.order.put(newSku, 1);
63        }
64        System.out.println(
65            theStockService.getStock().get(newSku).getCategory() + " SKU# " + newSku
66                + " is added to the order.");
67      } else {
68        System.out.println(
69            "Provided item with SKU# " + newSku + " is not in stock, cannot be added to the order.");
70      }
71    }

74     * take a SKU# as an argument and remove one instance of that item from the current order. It

79    public void removeOneFromOrder(String newSku) {
80      if (newSku == null) {
81        throw new NullValueException("Provided SKU# is null.");
82      }
83      if (newSku.length() == 0) {
84        throw new InvalidConstructorArgumentsException("Provided SKU# is invalid.");
85      }
86      if (!order.containsKey(newSku)) {
87        System.out.println("No item in the order matches provided SKU#.");
88      } else {
89        Integer newCount = Integer.valueOf(order.get(newSku).intValue() - 1);
90        this.order.put(newSku, newCount);
91        System.out.println("One item by SKU# " + newSku + " is removed from the order, " + newCount
92            + " of such item is left in the order.");
93        if (newCount == 0) {
94          order.remove(newSku);
95        } 
96      }
97    }

99    /**
100    * confirm that all items in the current order conform to the same measurement system. It should
101    * return a Boolean value ("true" if all items conform and "false" if the order includes mixed
102    * measurement systems). It should also print a report to the standard output giving useful
103    * information about the measurements (it is up to you to decide what this method should report).
104    *
105    * @return the boolean
106    */
107   public boolean validateMeasurementSystem() {
108     if (this.order == null) {
109       throw new NullValueException("Provided order is empty.");
110     }
111     if (order.isEmpty()) {
112       throw new InvalidHashMapKeyException("Provided order is empty.");
113     }
114     Boolean result = false;
115     String[] keyArray = this.order.keySet().toArray(new String[0]);
116     String firstMmeasurementSystem = this.theStockService.getStock().get(keyArray[0]).getSom()
117         .name();
118     for (String nextKey : keyArray) {
119       if (!firstMmeasurementSystem
120           .contentEquals(theStockService.getStock().get(nextKey).getSom().name())) {
121         result = false;
122       } else {
123         result = true;
124       }
125     }
126     if (result) {
127       System.out
128           .println("This order contains items only in " + firstMmeasurementSystem + " system.");
129     } else {
130       System.out.println("This order contains items of mixed measurement system.");
131     }
132     return result;
133   }

135   /**
136    * take a measurement system (metric or standard) as an argument and delete all items in the order
137    * that do not conform to that measurement system, leaving only those items in the order that do
138    * conform to the measurement system.
139    *
140    * @param measurementSystem the measurement system
141    */
142   public void filterByMeasurementSystem(String measurementSystem) {
143     if (this.order == null) {
144       throw new NullValueException("Provided order is null.");
145     }
146     if (measurementSystem == null) {
147       throw new NullValueException("Provided measurement system input is null.");
148     }
149     if (measurementSystem.length() == 0) {
150       throw new InvalidConstructorArgumentsException("Provided measurement system is invalid.");
151     }
152     String[] keyArray = this.order.keySet().toArray(new String[0]);
153     for (String nextKey : keyArray) {
154       if (!theStockService.getStock().get(nextKey).getSom().name()
155           .contentEquals(measurementSystem)) {
156         order.remove(nextKey);
157       }
158     }
159   }

161   /**
162    * check each item in the order for its category (e.g. "Precision Ball Bearings") and return the
163    * category if all items  share the same category, and return "Mixed" if the order contains items
164    * of different categories.
165    *
166    * @return the string
167    */
168   public String orderCategory() {
169     if (this.order == null) {
170       throw new NullValueException("Provided order is null.");
171     }
172     if (order.isEmpty()) {
173       throw new InvalidHashMapKeyException("Provided order is empty.");
174     }
175     String[] keyArray = order.keySet().toArray(new String[0]);
176     String firstItemCategory = theStockService.getStock().get(keyArray[0]).getCategory();
177     for (String nextKey : keyArray) {
178       if (!firstItemCategory
179           .contentEquals(theStockService.getStock().get(nextKey).getCategory())) {
180         return "Mixed";
181       }
182     }
183     return firstItemCategory;
184   }

186   /**
187    * print out (to standard output) a tabulated report of the current order listing the category,
188    * SKU#, and price of each item in the order, with the total cost at the bottom of the report.
189    */
190   public void printDocket() {
191     Integer totalPriceInCent = 0;
192     if (this.order == null) {
193       throw new NullValueException("Provided order is null.");
194     }
195     if (order.isEmpty()) {
196       throw new InvalidHashMapKeyException("Provided order is empty.");
197     }
198     String[] keyArray = order.keySet().toArray(new String[0]);
199     for (String nextKey : keyArray) {
200       System.out.println(
201           theStockService.getStock().get(nextKey).getCategory() + " " + nextKey + " "
202               + theStockService.getStock().get(nextKey).getPrice().printPrice());
203       totalPriceInCent += theStockService.getStock().get(nextKey).getPrice().getPriceInCent();
204     }
205     StringBuffer strbdr = new StringBuffer()
206         .append(theStockService.getStock().get(keyArray[0]).getPrice().getCurrency())
207         .append(((double) totalPriceInCent / 100.00));
208     System.out.print(strbdr.toString());
209   }

211   /**
212    * take two arbitrary stock item SKU#s and determine whether they represent a shaft/shaft-mounted
213    * pair of items for which the diameter of the shaft matches the "for shaft diameter" value of the
214    * shaft-mounted part. It should return a Boolean value representing whether the two items fit
215    * each other. If so, it should print a report to standard output following the given format.
216    *
217    * @param sku1 the sku 1
218    * @param sku2 the sku 2
219    * @return the boolean
220    */
221   public boolean checkFit(String sku1, String sku2) {
222     String category1;
223     String category2;
224     SystemOfMeasurement som1;
225     SystemOfMeasurement som2;
226     if (sku1 == null || sku2 == null) {
227       throw new NullValueException("Provided SKU# is null.");
228     }
229     if (sku1.length() == 0 || sku2.length() == 0) {
230       throw new InvalidConstructorArgumentsException("Provided SKU# is invalid.");
231     }
232     if (!theStockService.getStock().containsKey(sku1) && !theStockService.getStock()
233         .containsKey(sku2)) {
234       StringBuilder errorMsg3 = new StringBuilder();
235       errorMsg3.append("No item in the stock matches provided SKU# ").append(sku1).append(" and ")
236           .append(sku2).append(".");
237       System.out.println(errorMsg3.toString());
238     } else if (!theStockService.getStock().containsKey(sku1)) {
239       StringBuilder errorMsg1 = new StringBuilder();
240       errorMsg1.append("No item in the stock matches provided SKU# ").append(sku1).append(".");
241       System.out.println(errorMsg1.toString());
242     } else if (!theStockService.getStock().containsKey(sku2)) {
243       StringBuilder errorMsg2 = new StringBuilder();
244       errorMsg2.append("No item in the stock matches provided SKU# ").append(sku2).append(".");
245       System.out.println(errorMsg2.toString());
246     } else {
247       category1 = theStockService.getStock().get(sku1).getCategory();
248       category2 = theStockService.getStock().get(sku2).getCategory();
249       som1 = theStockService.getStock().get(sku1).getSom();
250       som2 = theStockService.getStock().get(sku2).getSom();
251       if ((!category1.contains("Rotary Shaft") && !category2
252           .contains("Rotary Shaft")) || (
253           category1.contains("Rotary Shaft") && category2.contains("Rotary Shaft"))) {
254         System.out.println("Not a shaft/shaft-mounted pair.");
255       } else if (!som1.equals(som2)) {
256         System.out.println("System of Measurement don't match.");
257       } else if (category1.contains("Rotary Shaft")) {
258         if (theStockService.getStock().get(sku1).getShaft()
259             .equals(theStockService.getStock().get(sku2).getShaft())) {
260           StringBuilder resMsg1 = new StringBuilder();
261           resMsg1.append(theStockService.getStock().get(sku2).getCategory()).append(" SKU# ")
262               .append(sku2).append(" for ")
263               .append(theStockService.getStock().get(sku2).getShaft().printMeasurement())
264               .append(" shaft fits on Rotary Shafts SKU# ").append(sku1);
265           System.out.println(resMsg1.toString());
266           return true;
267         } else {
268           StringBuilder resMsg2 = new StringBuilder();
269           resMsg2.append(theStockService.getStock().get(sku2).getCategory()).append(" SKU# ")
270               .append(sku2).append(" for ")
271               .append(theStockService.getStock().get(sku2).getShaft().printMeasurement())
272               .append(" shaft does not fit on Rotary Shafts SKU# ").append(sku1);
273           System.out.println(resMsg2);
274           return false;
275         }
276       }
277     }
278     return false;
279   }
