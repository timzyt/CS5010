1   package edu.neu.ccs.cs5010.assignment4.ridesharemodel.validators;
2   
3   import edu.neu.ccs.cs5010.assignment4.ridesharemodel.Registration;
4   import edu.neu.ccs.cs5010.assignment4.ridesharemodel.driver.Country;
5   
6   
7   /**
8    * The driver validator class.
9    */
10  public class DriverValidator implements MasterValidator {
11  
12    private Registration registration;
13    private TimeCalculator timeCalculator;
14    private Integer legalDrivingAge;
15    private Integer legalLicenseMonth;
16    private Integer zero = 0;
17  
18    /**
19     * Instantiates a new driver validator.
20     *
21     * @param registration the registration
22     */
23    public DriverValidator(Registration registration) {
24      this.registration = registration;
25      timeCalculator = new TimeCalculator();
26      legalDrivingAge = 21;
27      legalLicenseMonth = 6;
28    }
29  
30    /**
31     * Validates whether the prospective driver is over legal age (older than 21). If not, the
32     * prospective driver should not be accepted as a driver.
33     *
34     * @param registration the registration
35     * @return the boolean
36     */
37    public Boolean validateAge(Registration registration) {
38      return this.timeCalculator.calculateYear(registration.getDriver().getBirthday())
39          >= legalDrivingAge;
40    }
41  
42    /**
43     * Validates whether the prospective driver's name matches the name on the license. If not, the
44     * prospective driver should not be accepted as a driver.
45     *
46     * @param registration the registration.
47     * @return the boolean
48     */
49    public Boolean validateName(Registration registration) {
50      return registration.getDriver().getName()
51          .equals(registration.getDriver().getLicense().getName());
52    }
53  
54    /**
55     * Validates whether the prospective driver's birthday matches the birthday on the license. If
56     * not, the prospective driver should not be accepted as a driver.
57     *
58     * @param registration the registration
59     * @return the boolean
60     */
61    public Boolean validateBirthday(Registration registration) {
62      return registration.getDriver().getBirthday()
63          .equals(registration.getDriver().getLicense().getBirthday());
64    }
65  
66    /**
67     * Validates if the country of issuance on the prospective driver's license is either US or
68     * Canada.If not, the prospective driver should not be accepted as a driver.
69     *
70     * @param registration the registration
71     * @return the boolean
72     */
73    public Boolean validateCountryOfIssuance(Registration registration) {
74      Country issueCountry = registration.getDriver().getLicense().getIssueCountry();
75      return issueCountry.equals(Country.Canada) || issueCountry.equals(Country.US);
76    }
77  
78    /**
79     * Validates if the prospective driver's license is issued within 6 months. If yes, the
80     * prospective driver should not be accepted as a driver
81     *
82     * @param registration the registration
83     * @return the boolean
84     */
85    public Boolean validateLicenseDateOfIssuance(Registration registration) {
86      if (this.timeCalculator.calculateYear(registration.getDriver().getLicense().getIssueDate())
87          == zero) {
88        return
89            this.timeCalculator.calculateMonth(registration.getDriver().getLicense().getIssueDate())
90                >= legalLicenseMonth;
91      } else {
92        return true;
93      }
94    }
95  
96    /**
97     * Validates whether the prospective driver's license is expired. If yes, the prospective driver
98     * should not be accepted as a driver.
99     *
100    * @param registration the registration
101    * @return the boolean
102    */
103   public Boolean vaidateLicenseExpiration(Registration registration) {
104     return this.timeCalculator
105         .notPast(registration.getDriver().getLicense().getExpirationDate());
106   }
107 
108 
109   /**
110    * Performs overall validation for the prospective driver. If fails, the prospective driver should
111    * not be accepted as a driver.
112    *
113    * @return boolean value of the validation.
114    */
115   @Override
116   public boolean masterValidator(Registration registration) {
117     return this.validateAge(this.registration)
118         && this.validateName(this.registration)
119         && this.validateBirthday(this.registration)
120         && this.validateCountryOfIssuance(this.registration)
121         && this.validateLicenseDateOfIssuance(this.registration)
122         && this.vaidateLicenseExpiration(this.registration);
123   }
124 }
