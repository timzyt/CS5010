8   import java.util.HashSet;
9   
10  /**
11   * The driver pool validator class.
12   */
13  public class DriverPoolValidator implements MasterValidator {
14  
15    private DriverPool driverPool;
16  
17    /**
18     * Instantiates a new Driver pool validator.
19     *
20     * @param driverPool the driver pool
21     */
22    public DriverPoolValidator(DriverPool driverPool) {
23      this.driverPool = driverPool;
24    }
25  
26    /**
27     * Validate if the accepted driver is adding a vehicle that's already in the existing driver pool.
28     * If yes, return false, otherwise return true.
29     *
30     * @param registration the registration
31     * @return the boolean.
32     */
33    public Boolean validateUniqueness(Registration registration) {
34      Driver newDriver = registration.getDriver();
35      Vehicle newVehicle = registration.getVehicle();
36      if (driverPool.getDriverPoolDatabase().containsKey(newDriver) && driverPool
37          .getDriverPoolDatabase().get(newDriver).contains(newVehicle)) {
38        return false;
39      }
40      return true;
41    }
42  
43  
44    /**
45     * If the accepted driver has multiple vehicles, validates whether this driver is driving only one
46     * vehicle at the same time.If yes, return true, otherwise return false.
47     *
48     * @param registration the registration
49     * @return the boolean.
50     */
51    public Boolean validateDriverWithMultipleVehicles(Registration registration) {
52      Driver newDriver = registration.getDriver();
53      Driver newPersonInDriverSeat = registration.getVehicle().getPersonInDriverSeat();
54      HashSet<Driver> driverCheckSet = new HashSet<>();
55      //check if registered driver has no vehicle, if so return true;
56      if (this.driverPool.getDriverPoolDatabase().get(newDriver) == null
57          || this.driverPool.getDriverPoolDatabase().get(newDriver).isEmpty()
58          || newPersonInDriverSeat == null) {
59        return true;
60      } else {
61        for (Vehicle e : driverPool.getDriverPoolDatabase().get(newDriver)) {
62          if ((e.getPersonInDriverSeat().equals(newDriver))
63              || driverCheckSet.contains(e.getPersonInDriverSeat())) {
64            return false;
65          } else {
66            //if the personInDriverSeat is null, meaning nobody is driving this vehicle,
67            // so no need to add it into the checklist.
68            if (e.getPersonInDriverSeat() != null) {
69              driverCheckSet.add(e.getPersonInDriverSeat());
70            }
71          }
72        }
73      }
74      return true;
75    }
76  
77    /**
78     * Given a vehicle, find the physically identical vehicle in a given hashset, then return the
79     * person in the driving seat of the found vehicle.
80     *
81     * @param vehicleSet the vehicle set
82     * @param vehicle the vehicle
83     * @return the driver
84     */
85    //helper method to iterate through the vehicle set, and return the specific vehicle in the set
86    public Driver findDriverFromVehicleSet(HashSet<Vehicle> vehicleSet, Vehicle vehicle) {
87      for (Vehicle v : vehicleSet) {
88        if (v.equals(vehicle)) {
89          return v.getPersonInDriverSeat();
90        }
91      }
92      //if no identical vehicle is found in the set, return null.
93      return null;
94    }
95  
96    /**
97     * If one vehicle is registered under multiple drivers, validate that no different drivers should
98     * be driving the same vehicle at the same time, possibly in different directions. If no such
99     * conflict happens, return true, otherwise return false.
100    *
101    * @param registration the registration
102    * @return the boolean.
103    */
104   public Boolean validateSharedVehicle(Registration registration) {
105     Vehicle newVehicle = registration.getVehicle();
106     Driver newPersonInDriverSeat = registration.getVehicle().getPersonInDriverSeat();
107     if (newVehicle ==  null || newPersonInDriverSeat == null) {
108       return true;
109     }
110     for (Driver driver : driverPool.getDriverPoolDatabase().keySet()) {
111       //use findDriverFromVehicleSet to find same vehicle
112       if (findDriverFromVehicleSet(this.driverPool.getDriverPoolDatabase()
113           .get(driver),newVehicle) != null
114           && findDriverFromVehicleSet(this.driverPool.getDriverPoolDatabase()
115           .get(driver),newVehicle).equals(newPersonInDriverSeat)) {
116         return false;
117       }
118     }
119     return true;
120   }
121 
122   @Override
123   public boolean masterValidator(Registration registration) {
124     return validateUniqueness(registration)
125         && validateDriverWithMultipleVehicles(registration)
126         && validateSharedVehicle(registration);
127   }
128 }
