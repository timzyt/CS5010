116       for (Driver driver : this.insuredDrivers) {

119       }
120       insuredDriverList.delete(insuredDriverList.length() - 2, insuredDriverList.length());
121     }
122     return "Insurance{"
123         + "\nowner = " + owner.getName().getFirstName() + " " + owner.getName().getLastName()
124         + ",\nvehicle = " + vehicle.getColor() + " " + vehicle.getYear() + " " + vehicle.getMake()
