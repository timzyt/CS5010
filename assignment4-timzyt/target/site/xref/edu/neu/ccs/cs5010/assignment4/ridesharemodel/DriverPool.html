6   
7   import java.util.ArrayList;
8   import java.util.Collections;
9   import java.util.Comparator;
10  import java.util.HashMap;
11  import java.util.HashSet;
12  
13  /**
14   * Driver pool class. Serves as one of the ridershare models.
15   */
16  public class DriverPool {
17  
18    private HashMap<Driver, HashSet<Vehicle>> driverPoolDatabase;
19    private DriverPoolValidator driverPoolValidator;
20  
21    /**
22     * Instantiates a new Driver pool.
23     */
24    public DriverPool() {
25      driverPoolDatabase = new HashMap<Driver, HashSet<Vehicle>>();
26      driverPoolValidator = new DriverPoolValidator(this);
27    }
28  
29    /**
30     * Gets driver pool.
31     *
32     * @return the driver pool
33     */
34    public HashMap<Driver, HashSet<Vehicle>> getDriverPoolDatabase() {
35      return driverPoolDatabase;
36    }
37  
38    /**
39     * Add the driver and vehicle into the existing driver pool.
40     *
41     * @param registration the registration
42     */
43    public void addDriverVehicle(Registration registration) {
44      Driver newDriver = registration.getDriver();
45      Vehicle newVehicle = registration.getVehicle();
46      if (!this.driverPoolDatabase.keySet().contains(newDriver)) {
47        HashSet<Vehicle> newVehicleSet = new HashSet<Vehicle>();
48        if (newVehicle != null) {
49          newVehicleSet.add(newVehicle);
50        }
51        this.driverPoolDatabase.put(newDriver, newVehicleSet);
52      } else {
53        this.driverPoolDatabase.get(newDriver).add(newVehicle);
54      }
55    }
56  
57    /**
58     * Returns the result from the master validator whether the new registration was meeting all three
59     * requirements for the existing driver pool.
60     *
61     * @param registration the registration
62     * @return the boolean
63     */
64    public boolean validateRegistrationInPool(Registration registration) {
65      return this.driverPoolValidator.masterValidator(registration);
66    }
67  
68  
69    /**
70     * Concludes validations happen both in the registration and the existing driver pool. Add the
71     * provided driver into the existing driver database if passes the validation, otherwise print
72     * failure message.
73     *
74     * @param registration the registration
75     */
76    public void masterValidator(Registration registration) {
77      if (this.validateRegistrationInPool(registration)
78          && registration.validateRegistration()) {
79        this.addDriverVehicle(registration);
80        System.out.println("This driver is successfully added into the existing driver pool.");
81      } else {
82        System.out.println("Sorry this driver cannot be accepted at this time.");
83      }
84    }
85  
86    /**
87     * Find all the driver whose last name match the given last name.
88     *
89     * @param lastName provided last name.
90     * @return ArraryList of drivers as driverList.
91     */
92    public ArrayList<Driver> findDriverList(String lastName) {
93      ArrayList<Driver> driverList = new ArrayList<>();
94      for (Driver driver : this.driverPoolDatabase.keySet()) {
95        if (driver.getName().getLastName().equals(lastName)) {
96          driverList.add(driver);
97        }
98      }
99      Collections.sort(driverList, new Comparator<Driver>() {
100       @Override
101       public int compare(Driver obj1, Driver obj2) {
102         return obj1.getName().getFirstName().compareToIgnoreCase(obj2.getName().getFirstName());
103       }
104     });
105     return driverList;
106   }
107 
108   /**
109    * Find all vehicles matching every driver in the provided driver list.
110    *
111    * @param driverList the driver list
112    * @return HashMap of drivers and their registered vehicles
113    */
114   public HashMap<Driver, HashSet<Vehicle>> findFullList(ArrayList<Driver> driverList) {
115     HashMap<Driver, HashSet<Vehicle>> fullList = new HashMap<Driver, HashSet<Vehicle>>();
116     for (Driver driver : driverList) {
117       if (!this.driverPoolDatabase.get(driver).isEmpty()) {
118         fullList.put(driver, this.driverPoolDatabase.get(driver));
119       } else {
120         fullList.put(driver, null);
121       }
122     }
123     return fullList;
124   }
125 }
