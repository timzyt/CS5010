17    /**
18     * Print query result from the query based on provided last name.
19     *
20     * @param driverList the driver list
21     * @param fullList the full list of both drivers and vehicles
22     * @param fullRecordForMultipleDrivers the full DMV Record for multiple drivers
23     */
24    public void printQueryResult(ArrayList<Driver> driverList,
25        HashMap<Driver, HashSet<Vehicle>> fullList,
26        HashMap<Driver, HashSet<Record>> fullRecordForMultipleDrivers) {
27      if (driverList == null) {
28        System.out.println("No registered driver found.");
29      } else {
30        //when diverList isn't null,
31        // print each driver's last name and then first name such as "Smith, Anne"
32        for (Driver driver : driverList) {
33          System.out
34              .println(driver.getName().getLastName()
35                  + ", " + driver.getName().getFirstName());
36          //when fullList isn't null, print each driver's vehicle's information,
37          // in the order of year, color, make, model, licensePlate
38          if (fullList.get(driver) != null) {
39            for (Vehicle vehicle : fullList.get(driver)) {
40              System.out.println(
41                  "     " + vehicle.getYear() + " " + vehicle.getColor() + " " + vehicle.getMake()
42                      + " " + vehicle.getModel() + ", " + vehicle.getLicensePlate());
43            }
44          }
45          //when the Record isn't null, print each driver violation
46          if (fullRecordForMultipleDrivers.get(driver) != null) {
47            System.out.println("     Driving violations:");
48            for (Record record
49                : fullRecordForMultipleDrivers.get(driver)) {
50              System.out.println(record.getViolationType().toString());
51            }
52          }
53        }
54      }
55  
56    }
