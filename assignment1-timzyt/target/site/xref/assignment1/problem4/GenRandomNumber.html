12    /*****************************************************************************************
13     *  * GenRandomNumber method
14     *  * -generate a random number within range [min, max].
15     *  * -include min and max.
16     *
17     * @param min minimum value for this random number.
18     * @param max maximum value for this random number.
19     * @return a random number between the range of minimum and maximum.
20     */
21    public double genRandNum(Integer min, Integer max) {
22      Random random = new Random();
23      double randomNumber = min + (max - min + 1) * random.nextDouble();
24      return randomNumber;
25    }
