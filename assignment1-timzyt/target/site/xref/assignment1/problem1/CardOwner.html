101   @Override
102   public boolean equals(Object object) {
103     if (this == object) {
104       return true;
105     }
106     if (object == null || getClass() != object.getClass()) {
107       return false;
108     }
109     CardOwner cardOwner = (CardOwner) object;
110     return Objects.equals(firstName, cardOwner.firstName)
111         && Objects.equals(lastName, cardOwner.lastName)
112         && Objects.equals(address, cardOwner.address)
113         && Objects.equals(emailAddress, cardOwner.emailAddress);
