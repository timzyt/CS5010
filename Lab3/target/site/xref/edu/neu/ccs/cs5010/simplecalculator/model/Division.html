5   /**
6    * The Multiplication class.
7    */
8   public class Division extends BinOp {
9   
10    /**
11     * Constructs and instantiates a new division operand.
12     *
13     * @param left the left subexpression
14     * @param right the right subexpression
15     */
16    public Division(Expression left, Expression right) {
17      super(left, right);
18    }
19  
20    /**
21     * {@inheritDoc}.
22     */
23    @Override
24    public Val eval(Context ctx) {
25      Integer leftEval = this.left.eval(ctx).getVal();
26      Integer rightEval = this.right.eval(ctx).getVal();
27      if (rightEval == 0) {
28        throw new InvalidDenominatorException();
29      }
30  
31      return new Num(leftEval / rightEval);
32    }
33  
34    /**
35     * {@inheritDoc}.
36     */
37    @Override
38    public String asString() {
39      ExpressionVisitor newVisitor = new ExpressionVisitorImpl();
40      return newVisitor.visit(this);
41    }
42  }
