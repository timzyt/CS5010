1   package edu.neu.ccs.cs5010.simplecalculator.model;
2   
3   import java.util.Observable;
4   
5   /**
6    * abstract class Op.
7    */
8   abstract class Op extends Observable implements Expression {
9   
10    /**
11     * {@inheritDoc}.
12     */
13    public Integer evaluate() {
14      CtxHashMap newCtx = new CtxHashMap();
15      Num newNum = new Num(this.eval(newCtx).getVal());
16      return newNum.getVal();
17    }
18  
19    /**
20     * notify the Observer and print out to the console a trace of the evaluation.
21     */
22    public void setData(Expression exp) {
23      setChanged();
24      notifyObservers(exp);
25    }
26  
27    /**
28     * Return the next step evaluation of the given expression.
29     *
30     * @return expression.
31     */
32    public Expression next() {
33      return new Num(this.evaluate());
34    }
35  
36    /**
37     * {@inheritDoc}.
38     */
39    public void trace() {
40      Expression exp = this;
41      while (exp instanceof Op) {
42        this.setData(exp);
43        exp = ((Op) exp).next();
44      }
45    }
46  
47  }
