8   public class FileCombineUtil {
9   
10    /**
11     * Combine strings stored in a List into one complete string.
12     *
13     * @param stringList List containing strings.
14     * @return string value of combined strings.
15     * @throws Exception NullArgumentException
16     */
17    public String combineStringList(List<String> stringList) throws Exception {
18      StringBuilder strBdr = new StringBuilder();
19      if (stringList == null) {
20        throw new NullArgumentException("Provided list of string argument is null.");
21      }
22      if (stringList.size() == 0) {
23        throw new EmptyArgumentException("Provided list of string argument is empty.");
24      }
25      for (int i = 0; i < stringList.size(); i++) {
26        strBdr.append(stringList.get(i)).append(System.lineSeparator());
27      }
28      strBdr.deleteCharAt(strBdr.length() - 1);
29      return strBdr.toString();
30    }
31  
32    /**
33     * Combine strings stored in a Map into one complete string.
34     *
35     * @param stringMap Map containing strings.
36     * @return String value of combined strings.
37     * @throws Exception NullArgumentException
38     */
39    public String combineStringMap(Map<Integer, String> stringMap) throws Exception {
40      StringBuilder strBdr = new StringBuilder();
41      if (stringMap == null) {
42        throw new NullArgumentException("Provided list of string argument is null.");
43      }
44      if (stringMap.size() == 0) {
45        throw new EmptyArgumentException("Provided list of string argument is empty.");
46      }
47      for (int i = 0; i < stringMap.size(); i++) {
48        strBdr.append(stringMap.get(i));
49      }
50      return strBdr.toString();
51    }
