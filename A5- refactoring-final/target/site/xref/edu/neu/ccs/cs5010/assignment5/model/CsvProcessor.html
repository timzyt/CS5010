15  public class CsvProcessor{
16    BufferredReaderUtil newBru = new BufferredReaderUtil();
17    LineParser newLineparser = new LineParser();
18    Set<Map<String, String>> customerSet = new HashSet<>();
19    List<String> fileContent = new ArrayList<>();
20    List<String> header = new ArrayList<>();
21    List<String> currCustomerList = new ArrayList<>();
22    Integer zero = 0;
23  
24    /**
25     * Load csv file by file name, and return a Set of Maps where customer information is stored.
26     * @param fileName csv file name.
27     * @return Set a set of customer information.
28     */
29    public Set<Map<String, String>> loadCustomers(String fileName) {
30      fileContent = newBru.read(fileName);
31      header = newLineparser.parse(fileContent.get(zero));
32      for (int i = 1; i < fileContent.size(); i++) {
33        currCustomerList = newLineparser.parse(fileContent.get(i));
34        Map<String, String> currCustomerMap = new HashMap<>();
35        for (int j = 0; j < header.size(); j++) {
36          currCustomerMap.put("[[" + header.get(j) + "]]", currCustomerList.get(j));
37        }
38        customerSet.add(currCustomerMap);
39      }
40      return customerSet;
41    }
42  }
