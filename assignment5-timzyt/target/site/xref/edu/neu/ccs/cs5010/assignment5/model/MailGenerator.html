26    public void run(String mailType, String templateName, String csvFileName, String outputDir) throws IOException {
27  //    String cachedTemplate = newTemplateReader.run(templateName);
28  //    System.out.println(cachedTemplate);
29      keys = newCsvReader.getKeys(csvFileName);
30      System.out.println(keys.get(0));
31      System.out.println(System.getProperty("user.dir"));
32      BufferedReader inputFile = null;
33      try {
34        inputFile = new BufferedReader(new FileReader(csvFileName));
35  
36        String csvLine;
37        inputFile.readLine();
38  
39        while ((csvLine = inputFile.readLine()) != null) {
40          System.out.println("Read : " + csvLine);
41          currCsvLine = newLineParser.parse(csvLine);
42  
43  //          for (String value : currCsvLine) {
44  //            System.out.println("value in currVsvLine is: " + value);
45  //          }
46          // reset outputFileName to the base case
47          String outputFileName = mailType;
48          StringBuilder group0 = new StringBuilder();
49          Pattern re0;
50          String template = newTemplateReader.run(templateName);
51  
52          for (int i = 0; i < keys.size(); i++) {
53            group0.append("\\[\\[").append(keys.get(i).replace("\"", "")).append("\\]\\]");
54            re0 = Pattern.compile(group0.toString());
55            Matcher m = re0.matcher(template);
56            if (m.find()) {
57              template = template.replace(m.group(),
58                  currCsvLine.get(i).replace("\"", ""));
59            }
60            group0.setLength(0);
61          }
62          System.out.println("template after replacement is: \r\n" + template);
63          System.out.println(currCsvLine.get(0));
64          outputFileName = outputFileName + "_"
65              + currCsvLine.get(0).replace("\"", "") + " "
66              + currCsvLine.get(1).replace("\"", "") + ".txt";
67  
68          BufferedWriter outputFile = new BufferedWriter(
69              new FileWriter(outputDir + File.separator + outputFileName));
70          outputFile.write(template);
71          outputFile.flush();
72          outputFile.close();
73        }
74  
75  
76      } catch (FileNotFoundException fnfe) {
77        System.out.println("*** OUPS! A file was not found : " + fnfe.getMessage());
78        fnfe.printStackTrace();

82      }
