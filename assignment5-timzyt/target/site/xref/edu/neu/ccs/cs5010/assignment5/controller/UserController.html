11  
12  /**
13   * The User controller class.
14   */
15  public class UserController {
16  
17    /**
18     * The main class, entry point of application.
19     *
20     * @param args the input arguments from user input.
21     */
22    public static void main(String[] args) {
23      String email = "--email";
24      String letter = "--letter";
25      String emailTemplate = "--email-template";
26      String letterTemplate = "--letter-template";
27      String outputDir = "--output-dir";
28      String csvFile = "--csv-file";
29  
30      HashMap<String, String> cmdPairs = new HashMap<>();
31      cmdPairs.put(email, emailTemplate);
32      cmdPairs.put(letter, letterTemplate);
33  
34      System.out.println(System.getProperty("user.dir"));
35  
36      try (
37          Scanner cmdInput = new Scanner(System.in, "UTF-8")) {
38        CmdProcessor newCmdProcessor = new CmdProcessor();
39        //boolean variable to indicate whether input is found having error
40        boolean checkInput = false;
41        String newCmdInputStr = cmdInput.nextLine();
42        final String[] newCmdInput = newCmdInputStr.split(" ");
43        if (!newCmdInputStr.contains(email) && !newCmdInputStr.contains(letter)) {
44          System.err.printf("Error: Missing command %s or %s.%n%n", email, letter);
45          System.exit(0);
46        }
47        if (!newCmdInputStr.contains(outputDir)) {
48          System.err.printf("Error: Missing command %s.%n%n", outputDir);
49          System.exit(0);
50        }
51        if (!newCmdInputStr.contains(csvFile)) {
52          System.err.printf("Error: Missing command %s.%n%n", csvFile);
53          checkInput = true;
54          System.exit(0);
55        }
56        for (Map.Entry<String, String> entry : cmdPairs.entrySet()) {
57          String key = entry.getKey();
58          if (newCmdInputStr.contains(key) && !newCmdInputStr.contains(cmdPairs.get(key))) {
59            System.err.printf("Error: %s provided but no %s was given.%n%n", key,
60                cmdPairs.get(key));
61            checkInput = true;
62  
63          }
64        }
65        if (newCmdInputStr.contains(email) && newCmdInputStr.contains(letter)) {
66          System.err.println("Error: Conflicting command line argument was provided.%n%n");
67          checkInput = true;
68          System.exit(0);
69        }
70        if (checkInput) {
71          CommandLine.usage(newCmdProcessor, System.out);
72          System.exit(0);
73        }
74  
75        new CommandLine(newCmdProcessor).parse(newCmdInput);
76        newCmdProcessor.runMailGenerator();
77  
78  
79      } catch (RuntimeException rte) {
80        System.out.println(rte.getMessage());
81        rte.printStackTrace();
82      } catch (Exception e) {
83        System.out.println(e.getMessage());
84        e.printStackTrace();
85      }
86    }
87  }
