21    /**

24     */
25    @Option(names = "--email", description = "only generate emails messages")
26    public boolean createEmail;

28    /**
29     * The email template option.

32    @Option(arity = "1", names = "--email-template", paramLabel = "email template", description = "accept a filename that holds the email template.\r\n\tRequired if --email is used.")

34  
35    /**
36     * The create letter option.
37     * Returns a boolean value whether email creation has been triggered.
38     */
39    @Option(names = "--letter", description = "only generate letters")
40    public boolean createLetter;

42    /**
43     * The letter template option.
44     * Receives a string as the letter template name.
45     */
46    @Option(arity = "1", names = "--letter-template", paramLabel = "letter template", description = "accept a filename that holds the letter template.\r\n\tRequired if --letter is used.")
47    public String letterTemplate;
48  
49    /**
50     * The output directory option.
51     * Receives a string as the output sub-directory.
52     */
53    @Option(arity = "1", names = "--output-dir", paramLabel = "output directory", description = "accept the name of a folder, all output is placed in this folder.")

56    /**

70    /**
71     * Run mail generator method.
72     *
73     * @throws IOException the io exception
74     */
75    //run mailGenerator, pass in the arguments
76    public void runMailGenerator() throws IOException {
77      MailGenerator newMailGenerator = new MailGenerator();
78      try {
79        if (createEmail) {
80          newMailGenerator.run("email", emailTemplate, csvFile, outputDir);
81        }
82        if (createLetter) {
83          newMailGenerator.run("letter", letterTemplate, csvFile, outputDir);
84        }
85  
86      } catch (Exception e) {
87        System.out.println(e.getMessage());
88        e.printStackTrace();
89      }
90    }
