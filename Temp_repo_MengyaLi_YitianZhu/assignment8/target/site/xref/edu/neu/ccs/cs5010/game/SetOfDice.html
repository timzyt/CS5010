46    /**
47     * Get the dice set.
48     *
49     * @return the dice [ ]
50     */
51    public Dice[] getDiceSet() {
52      Dice[] copyArray = new Dice[this.diceSet.length];
53      for (int i = 0; i < diceSet.length; i++) {
54        copyArray[i] = new Dice();
55        copyArray[i].setValue(diceSet[i].getValue());
56      }
57      return copyArray;
58    }
59  
60    /**
61     * Instantiates a new Set of dice.
62     */
63    public SetOfDice() {
64      diceSet = new Dice[TOTAL_NUM_OF_DICE];
65      for (int i = ZERO; i < TOTAL_NUM_OF_DICE; i++) {
66        diceSet[i] = new Dice();
67      }
68      //rand = new Random();
69    }
70  
71    /**
72     * Roll all dice.
73     */
74    public void rollAllDice() {
75      for (int i = ZERO; i < TOTAL_NUM_OF_DICE; i++) {
76        this.diceSet[i].roll();
77      }
78    }
79  
80    /**
81     * Roll selected dice.
82     *
83     * @param diceSelection the dice selection
84     */
85    public void rollSelectedDice(int[] diceSelection) {
86      for (int j = ZERO; j < TOTAL_NUM_OF_DICE; j++) {
87        if (!this.contains(diceSelection, j + 1)) {
88          this.diceSet[j].roll();
89        }
90      }
91  
92    }
93  
94    /**
95     * Find if target item is in the array.
96     *
97     * @param arr the arr
98     * @param key the index for the target
99     * @return the boolean
100    */
101   public static boolean contains(final int[] arr, final int key) {
102     return Arrays.stream(arr).anyMatch(index -> index == key);
103   }
104 
105   /**
106    * Gets count of given value.
107    *
108    * @param val the value
109    * @return the count of value
110    */
111   public int getCountOfValue(int val) {
112     int counter = ZERO;
113     for (int i = ZERO; i < TOTAL_NUM_OF_DICE; i++) {
114       if (this.diceSet[i].getValue() == val) {
115         counter++;
116       }
117     }
118     return counter;
119   }
120 
121   /**
122    * Gets total value.
123    *
124    * @return the total value
125    */
126   public int getTotalValue() {
127     int total = ZERO;
128     for (int i = ZERO; i < TOTAL_NUM_OF_DICE; i++) {
129       total += this.diceSet[i].getValue();
130     }
131     return total;
132   }

135    * Get an array of dice values.

137    * @return the integer array

139   public int[] getDiceValues() {
140     int[] diceArray = new int[FIVE];
141     for (int i = ZERO; i < TOTAL_NUM_OF_DICE; i++) {
142       diceArray[i] = diceSet[i].getValue();
143     }
144     return diceArray;
145   }

147   /**
148    * Check if the provided set contains valid indices.
149    *
150    * @param diceChoice string input of the dice choice
151    * @return boolean
152    */
153   public boolean checkValidSelection(String diceChoice) {
154     try {
155       int[] keep = Arrays.stream(diceChoice.split(" ")).mapToInt(Integer::parseInt).toArray();
156       for (int i : keep) {
157         if (i < 0 || i >= 6) {
158           return false;
159         }
160       }
161       return true;
162     } catch (Exception e) {
163       return false;
164     }
165 
166   }

175       this.diceSet[i].setValue(sod[i]);

180   public String toString() {

183         + this.diceSet[TWO].getValue() + " "
