93    /**
94     * Given the threshold, find the peak phase from the bulk result generated by the map builder.
95     *
96     * @param threshold the threshold
97     * @return the peak phase data represented in a Map data structure.
98     * @throws Exception IllegalThresholdException
99     */
100   public Map<Long, ArrayList<Long>> findPeakPhase(Long threshold) throws Exception {
101     Iterator<Map.Entry<Long, ArrayList<Long>>> setIt = result.entrySet().iterator();
102     while (setIt.hasNext()) {
103       Map.Entry<Long, ArrayList<Long>> pair = setIt.next();
104       ArrayList<Long> currValue = pair.getValue();
105       if (currValue.get(INDEX_OF_COUNT) >= threshold) {
106         peakStartKey = pair.getKey();
107         break;
108       }
109     }
110     if (peakStartKey == null) {
111       System.out.println("peakStartKey is null.");
112       throw new IllegalThresholdException("Provided threshold is too high.");
113     }
114 
115     Iterator<Map.Entry<Long, ArrayList<Long>>> setIt2 = result.descendingMap().entrySet()
116         .iterator();
117     while (setIt2.hasNext()) {
118       Map.Entry<Long, ArrayList<Long>> pair = setIt2.next();
119       ArrayList<Long> currValue = pair.getValue();
120       if (currValue.get(INDEX_OF_COUNT) >= threshold) {
121         peakEndKey = result.lowerKey(pair.getKey());
122         break;
123       }
124     }
125     Map<Long, ArrayList<Long>> subResult = result.subMap(peakStartKey, peakEndKey);
126     this.thresholdResult = subResult;
127     return thresholdResult;
128   }

131   /**
132    * Add.
133    *
134    * @param partition the partition
135    */
136   public void addThroughputPartition(ConcurrentNavigableMap partition) {
137     Iterator<Long> iter = partition.keySet().iterator();
138     while (iter.hasNext()) {
139       Long key = iter.next();
140       ArrayList<Long> value = (ArrayList<Long>) partition.get(key);
141       if (this.throughputMap.containsKey(key)) {
142         this.throughputMap.get(key).addAll(value);
143       } else {
144         this.throughputMap.put(key, value);
145       }
146     }
147   }

150    * Add.

152    * @param partition the partition

154   public void addLatencyPartition(ConcurrentNavigableMap partition) {
155     Iterator<Long> iter = partition.keySet().iterator();
156     while (iter.hasNext()) {
157       Long key = iter.next();
158       Integer value = (Integer) partition.get(key);
159       if (this.latencyMap.containsKey(key)) {
160         Integer newValue = this.latencyMap.get(key) + value;
161         this.latencyMap.put(key, newValue);
162       } else {
163         this.latencyMap.put(key, value);
164       }
165     }
166   }

168   /**
169    * Process threshold result. As byproduct calculates 5 percentile throughput result calculates 99
170    * percentile latency result
171    *
172    * @param thresholdResult the threshold result
173    */
174   public void analyzePeakPhase(Map<Long, ArrayList<Long>> thresholdResult) {
175     TreeMap<Long, ArrayList<Long>> copyMap = new TreeMap<>(thresholdResult);
176     GetPercentile getThroughputPct = new GetPercentile((long) copyMap.size(), FIFTH_PERCENTILE);
177     GetPercentile getLatencyPct = new GetPercentile((long) copyMap.size(), NINTYNINE_PERCENTILE);
178     Iterator<Map.Entry<Long, ArrayList<Long>>> setIt = copyMap.entrySet().iterator();
179     while (setIt.hasNext()) {
180       Map.Entry<Long, ArrayList<Long>> pair = setIt.next();
181       ArrayList<Long> value = pair.getValue();
182       Long currThroughput = value.get(INDEX_OF_COUNT);
183       Long currLatency = value.get(INDEX_OF_MEAN);
184       thresholdThroughputSum += currThroughput;
185       thresholdLatencySum += currLatency;
186       thresholdPeakThroughput = Math.max(thresholdPeakThroughput, currThroughput);
187       getThroughputPct.add(currThroughput);
188       getLatencyPct.add(currThroughput);
189     }
190     thresholdMeanThroughput = thresholdThroughputSum / this.thresholdResult.size();
191     thresholdMeanLatency = thresholdLatencySum / this.thresholdResult.size();
192     fifthPctThroughput = getThroughputPct.getPercentile();
193     nintyniePctLatency = getLatencyPct.getPercentile();
194   }

197   /**
198    * Analyze all data and calculates requirement metrics.
199    */
200   public void analyzeAllData() {
201 
202     //this.throughputMap = this.throughputMap.descendingMap();
203     //calculate percentile
204     Long mapSize = Long.valueOf(throughputMap.size());
205     //construct the percentile calculation object based on the bucket results
206     GetPercentile getPct = new GetPercentile(mapSize, percentile);
207 
208     Iterator<Map.Entry<Long, ArrayList<Long>>> setIt = this.throughputMap.entrySet().iterator();
209     while (setIt.hasNext()) {
210       Map.Entry<Long, ArrayList<Long>> pair = setIt.next();
211       Long key = pair.getKey();
212       ArrayList<Long> value = pair.getValue();
213       this.maxSec = Long.max(key, this.maxSec);
214       this.minSec = Long.min(key, this.minSec);
215       Long numLatency = Long.valueOf(value.size());
216       Long sumLatency = 0L;
217       for (Long d : value) {
218         sumLatency += d;
219         getPct.add(d);
220       }
221       ArrayList<Long> newValue = new ArrayList<Long>();
222       newValue.add(numLatency);
223       newValue.add(sumLatency / numLatency);
224       this.totalNumLatency += numLatency;
225       this.totalSumLatency += sumLatency;
226       this.meanLatency = this.totalSumLatency / this.totalNumLatency;
227       this.result.put(key, newValue);
228     }
229 
230     this.percentileResult = getPct.getPercentile();
231   }
